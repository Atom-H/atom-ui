# Dialog

弹出层, 通过半透明遮罩, 用于突出提醒.

# 基本使用

{
    data(){
        return {isShow: false};
    }
}

<a-dialog :is-show.sync="isShow">
    <a-cell>正常</v-cell>
    <a-cell>困难</v-cell>
    <a-cell>地狱</v-cell>
    <a-button type="primary" :is-block="true" class="gutter-top">开始游戏</a-button>
    <a-button :is-ghost="true" :is-block="true" @click="isShow=false" class="gutter-top">离开</a-button>
</v-dialog>
