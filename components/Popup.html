# Popup

弹出层

# 基本使用

//从上方滑入
{
    data() {
        return {
            from: 'top',
            isShow: false
        };
    }
}

    <a-popup :is-show.sync="isShow" :from="from">
        <a-cell>正常</a-cell>
        <a-cell>困难</a-cell>
        <a-cell>地狱</a-cell>
        <div class="fill">
            <a-button type="primary" :is-block="true">开始游戏</a-button>
            <a-button :is-ghost="true" :is-block="true" @click="isShow=false" class="gutter-top">离开</a-button>
        </div>
    </a-popup>
