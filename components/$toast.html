# $toast

# 基本使用

// 默认3秒后自动关闭,
this.$toast('3秒后自动关闭');

// 支持4种情景: success / warning / danger / loading
this.$toast('我支持情景色', {type:'success'});

// 关闭, 注意一个页面同时只有一个toast
this.$toast.close();
